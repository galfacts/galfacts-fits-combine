200 void DesktopConnector::refreshViewNow(IView *view)
201 {
202 
203  qDebug() << "refreshViewNow " << view->name();
204 
205  ViewInfo * viewInfo = findViewInfo( view-> name());
206  if( ! viewInfo) {
207  // this is an internal error...
208  qCritical() << "refreshView cannot find this view: " << view-> name();
209  return;
210  }
211  // get the image from view
212  const QImage & origImage = view-> getBuffer();
213  // scale the image to fit the client size
214  QImage destImage = origImage.scaled(
215  viewInfo->clientSize, Qt::KeepAspectRatio,
216  // Qt::SmoothTransformation);
217  Qt::FastTransformation);
218  // calculate the offset needed to center the image
219  int xOffset = (viewInfo-> clientSize.width() - destImage.size().width())/2;
220  int yOffset = (viewInfo-> clientSize.height() - destImage.size().height())/2;
221  QImage pix( viewInfo->clientSize, QImage::Format_ARGB32_Premultiplied);
222  pix.fill( qRgba( 0, 0, 0, 0));
223  QPainter p( & pix);
224  p.setCompositionMode( QPainter::CompositionMode_Source);
225  p.drawImage( xOffset, yOffset, destImage );
226 
227  // remember the transformations we did to the image in the viewInfo so that we can
228  // properly translate mouse events etc
229  viewInfo-> tx = LinearMap1D( xOffset, xOffset + destImage.size().width()-1,
230  0, origImage.width()-1);
231  viewInfo-> ty = LinearMap1D( yOffset, yOffset + destImage.size().height()-1,
232  0, origImage.height()-1);
233 
234  emit jsViewUpdatedSignal( view-> name(), pix);
235 }
